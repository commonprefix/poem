\begin{algorithm}[h]
  \caption{\label{alg.maxvalid} The maxvalid algorithm}
  \begin{algorithmic}[1]
  \Function{$\maxvalid_{\mathcal{G}}$}{$\overline{C}$}
    \Let{C_\mmax}{[\,]}
    \Let{\maxwork}{0}
    \For{$C \in \overline{C}$}
      \If{$\lnot\validate_{\mathcal{G}}(C)$}
        \Continue
      \EndIf

      \Let{\thiswork}{\awork(C)} \Comment{Computed as $\sum_{B \in C} -\lg H(B)[{:}{\kappa}]$}
      \If{$\thiswork > \maxwork$} \label{alg-maxvalid:validate}
        \Let{C_\mmax}{C}
        \Let{\maxwork}{\thiswork}
      \EndIf
    \EndFor
    \State\Return{$C_\mmax$}\label{alg-maxvalid:return}
  \EndFunction
  \end{algorithmic}
\end{algorithm}

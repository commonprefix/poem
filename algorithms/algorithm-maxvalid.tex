\begin{algorithm}[H]
  \caption{\label{alg.maxvalid} The maxvalid algorithm}
  \begin{algorithmic}[1]
  \Function{$\maxvalid_{\mathcal{G}}$}{$\overline{C}$}
    \Let{C_\mmax}{\epsilon}
    \For{$C \in \overline{C}$}
      \If{$\validate_{\mathcal{G}}(C) \land \work(C) > \work(C_\mmax)$} \label{alg-maxvalid:validate}
        \Let{C_\mmax}{C}
      \EndIf
    \EndFor
    \State\Return{$C_\mmax$}\label{alg-maxvalid:return}
  \EndFunction
  \end{algorithmic}
\end{algorithm}

\begin{algorithm}[H]
  \caption{\label{alg.pow} The Proof-of-Work discovery algorithm}
  \begin{algorithmic}[1]
    \Function{$\pow_{H,T,q}$}{$x, h$}
        \State{$ctr \getsrandomly \{0,1\}^\kappa$}
        \For{$i \gets 1 \text{ to } q$}
            \Let{B}{h \concat x \concat ctr}
            \If{$H(B)[{:}{\kappa}] \leq T$}
                \State\Return{$B$}
            \EndIf
            \Let{ctr}{ctr + 1}
        \EndFor
        \State\Return{$\bot$}
    \EndFunction
  \end{algorithmic}
\end{algorithm}
